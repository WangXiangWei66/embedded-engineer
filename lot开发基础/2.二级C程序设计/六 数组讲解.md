int i=0, j=0;
//循环统计的下标
*a=0;//初始化a字符统计的个数

*c=0;//初始化c字符统计的个数

for(i=0; i<M; i++)//行
{
for(j=0; j<M; j++)//列
{
if(t[i][j]==’A’)//字符是a,计数
(*a)++;
if(t[i][j]==’C’)//字符是c,计数
(*c)++;
}
}
}

{  for( j=0; j<M; j++)
{  s[i][j]=65+rand()%12; printf( "%c ",s[i][j]); }
printf("\n");
}
}

{  char  a[M][M];
int   x, y;
get (a);
fun ( a, &x,&y );
printf("A = %d   C = %d\n",x,y);
NONO();
}
